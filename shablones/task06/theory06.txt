from turtle import *   # библиотека для черепахи

screensize(3000, 3000)   # размер вашего окна
tracer(0)   # скип анимок

s = 25   # Для размера
down()
for _ in range(n): #   повторить n раз
    rt(n)   # направо градусов
    fd(n * s)   # вперед на
    lt(n)   # вперед да
    bk(n * s)   # назад
up()   # поднять перо

for x in range(-50, 50):   # размер по иксу для точек
    for y in range(-50, 50):   # размер по y для точек
        goto(x * s, y * s)   # тут ставятся точки (черепаха перемещается)
        dot(3, 'red')   # настройка точек (черепаха ставит под себя точки)
done()   # ну усё
